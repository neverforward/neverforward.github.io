1006 - 打印星号三角形

题目描述

打印星号三角形。

输入

输入只有一行，包括1 个整数 𝑛，𝑛代表行数。

输出

输出一个星号三角形，行数为𝑛。

code:


#include<iostream>
using namespace std;

int main(){
    int n;
    cin>>n;
    for(int i=1; i<=n; i++){
        for(int j=1; j<=3; j++){
            for(int k=1; k<=n-i; k++){
                cout<<" ";
            }
            for(int k=1; k<=2*i-1; k++){
                cout<<"*";
            }
            for(int k=1; k<=n-i; k++){
                cout<<" "; 
            }
        }
        cout<<endl;
    }
    return 0;
}

1007 - 统计大写英文字母的个数

题目描述

算算以 '.' 结束的一串字符中含有多少个大写的英文字母。

输入

输入一串字符（长度不超过 80 ），以 '.'结束。

输出

输出一行，即这串字符中大写字母的个数。

code:


#include<iostream>
#include<string>
using namespace std;

int main() {
    int sum = 0;
    string input;
    getline(cin, input);
    for (int i = 0; i <= input.length(); i++) {
        if (input[i] >= 'A' && input[i] <= 'Z') {
            sum++;
        }
    }
    cout << sum << endl;
    return 0;
}

1009 - 数组逆序

题目描述

给你𝑛个整数，将其逆序输出。

输入

第一行一个整数 𝑛(3≤𝑛≤100)代表数的个数。第二行 𝑛个整数（空格隔开）

输出

n 个整数（空格隔开）。

code:


#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;

int a[1000];
int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}
	reverse(a, a + n);
	for (int i = 0; i < n; i++) {
		cout << a[i] <<' ';
	}
	return 0;
}

1010 - 数组元素的排序

题目描述

对数组的元素按从小到大进行排序。

输入

第一行一个整数 𝑛(5≤𝑛≤10)代表数的个数。第二行有𝑛个整数

输出

输出排序后的数组。

code:


#include <cstdio>
#include <algorithm>
using namespace std;

const int N = 1e6 + 10;
int a[N], n;
int jzd(int l,int r) {
	int p=a[l];
	int i=l,j=r;
	while(i<j){
		while(a[j]>=p&&i<j){
			j--;
		}
		while(a[i]<=p&&i<j){
			i++;
		}
		if(i<j){
			swap(a[i],a[j]);
			
		}
	}
	swap(a[l],a[i]);
	return i;
}
void Qsort(int l,int r) {
	if(l>r){
		return ;
	}
	int mid=jzd(l,r);
	Qsort(l,mid-1);
	Qsort(mid+1,r);
	
}
int main() {
	scanf("%d", &n);
	int i,j,t;
	for (i = 0; i < n; i++) {
		scanf("%d", &a[i]);
	}
	Qsort(0,n-1);
	for (int i = 0; i < n; i++) {
		printf("%d ", a[i]);
	}
	return 0;
}
